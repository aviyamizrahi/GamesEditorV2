EditGame.razor 
@page "/EditGame/{gameIdToEdit:int}"
@using AuthTemplate.Shared.Games
@using AuthTemplate.Shared.Games;
@inject HttpClient Http;
@inject NavigationManager Nav;
<h3>EditGame</h3>

<p>הוספת משחק חדש</p>
<EditForm Model="gameToEdit" OnValidSubmit="CreateGame">
    <DataAnnotationsValidator />
    <div>
        <label>שם משחק: 
            <InputText @bind-Value="gameToEdit.GameName" placeholder="שם משחק..."  @bind-Value:after="CheckCanSave" />
        </label>
    </div>
    <br/>
    <div>
        <label>זמן למיון פריט בכל תור</label>
        <InputNumber @bind-Value="gameToEdit.RoundTime"
                     step="5"
                     min="20"
                     max="60"
                     disabled="@noLimit" @oninput="CheckCanSave" />
        <b> או </b>
        <label>
            @* מריץ את הערך החדש ולא הישן *@
            <InputCheckbox @bind-Value="noLimit" @bind-Value:after="OnNoLimitChanged"  @bind-Value:after="CheckCanSave"/>
            ללא הגבלה
        </label>
    </div>
    <br/>
    <div>
        <label>הנחיות העורך: 
            <InputTextArea @bind-Value="gameToEdit.Instructions"  @bind-Value:after="CheckCanSave" />
        </label>
    </div>
    <ValidationMessage For="@(() => gameToEdit.GameName)" />

    <br />
    <input type="submit" value="שמירה" disabled="@(!context.Validate() || !canSave)" />
</EditForm> <br/>
<p>@msg</p>


@code {
    GameToAddDTO gameToEdit = new GameToAddDTO();
    bool noLimit = false;
    string msg = "";
    GameToAddDTO originalGame;
    bool canSave = false; // כפתור השמירה מתחיל כלא פעיל והופך להיות פעיל רק במידה ויש שינוי

    
    [Parameter]
    public int gameIdToEdit { get; set; }
    // פרמטר שמגיע מהכפתור בעמוד ״המשחקים שלי״
    // קולטות את המשתנה גם פה וגם דרך הנתיב כint

    protected override async Task OnParametersSetAsync()
    // כאשר הוגדר פרמטר וקיבלנו מספר דרך הנתיב
    {
        if (gameIdToEdit == 0) // אם זה משחק חדש
        {
        gameToEdit.ID = gameIdToEdit; 
        }
        else if (gameIdToEdit > 0) // אם זה משחק קיים 
        {
            var gameFromServer = await Http.GetAsync($"api/Games/GetGame/{gameIdToEdit}");
            if (gameFromServer.IsSuccessStatusCode)  // הבאת הנתונים של המשחק מבסיס הנתונים
            {
                gameToEdit = await gameFromServer.Content.ReadFromJsonAsync<GameToAddDTO>();
                if (gameToEdit.RoundTime == 0)
                {
                    noLimit = true; // אם הנתון של ללא הגבלה מסומן אז גם הצ׳ק בוקס צריך להופיע מסומן
                }

                originalGame = new GameToAddDTO 
                    //  כדי לבדוק אם יש שינויים בתיבות הטקסט נעשה מופע של משחק מקורי, בעתיד נשתמש בו גם ל״ביטול שינויים״
                {
                    GameName = gameToEdit.GameName,
                    RoundTime = gameToEdit.RoundTime,
                    Instructions = gameToEdit.Instructions
                };
            }
        }
    }
    
    void CheckCanSave()
    {
        // פונקציה שבודקת האם בוצעו שינויים בתיבות הטקסט / מספר 
        // במידה ובוצעו שינויים כפתור שמירה יהיה פעיל ובמידה ולא יהיה לא פעיל
        if (gameToEdit.GameName != originalGame.GameName || gameToEdit.RoundTime != originalGame.RoundTime ||
            gameToEdit.Instructions != originalGame.Instructions)
        {
            canSave = true;
        }
        else
        {
            canSave = false;
        }
        
        
    }
    
    void OnNoLimitChanged()
    {
        if (noLimit)
        {
        gameToEdit.RoundTime = 0;
        }
        else
        {
        gameToEdit.RoundTime = 30; // ברירת מחדל כשמבטלים
        }
    }


    async Task CreateGame()
    // בעת לחיצה על כפתור ״שמירה״ הפונקציה תפעל
    {
        if (gameToEdit.ID == 0)
            // אם זה משחק חדש תקרא לפונקציה בשרת שיוצרת משחק חדש
        {
            var newGameRes = await Http.PostAsJsonAsync("api/Games/addGame", gameToEdit);
            if (newGameRes.IsSuccessStatusCode)
            {
                GameToTableDTO newGame = newGameRes.Content.ReadFromJsonAsync<GameToTableDTO>().Result;
                gameToEdit.ID = newGame.ID;
                msg = "המשחק נוצר בהצלחה";
            }
        }
        else if (gameToEdit.ID > 0)
            // אם זה משחק קיים תעשכן את הפרטים בהתאם
        {
            var editGameRes = await Http.PostAsJsonAsync("api/Games/UpdateGame", gameToEdit);
            if (editGameRes.IsSuccessStatusCode)
            {
                msg = "המשחק עודכן בהצלחה";
            }
            
        }
    }


}
